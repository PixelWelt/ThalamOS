32def setup() -> None:
33 """
34 Sets up the database by creating the 'storage' table and a trigger for automatic
35 updating of the 'modification_time' column.
36 The 'storage' table contains the following columns:
37 - id: INTEGER, primary key, autoincrement
38 - position: INTEGER
39 - type: TEXT
40 - name: TEXT
41 - info: TEXT
42 - modification_time: TIMESTAMP, defaults to the current timestamp
43 The trigger 'update_modification_time' ensures that the 'modification_time' column
44 is automatically updated to the current timestamp whenever a row in the 'storage'
45 table is updated.
46 Commits the changes to the database and prints a confirmation message.
47 """
48 cursor.execute("""
49 CREATE TABLE IF NOT EXISTS storage (
50 id INTEGER PRIMARY KEY AUTOINCREMENT,
51 position INTEGER,
52 type TEXT,
53 name TEXT,
54 info TEXT,
55 modification_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP
56 );
57 """)
58
59 # trigger for automatic update of modification_time
60 cursor.execute("""
61 CREATE TRIGGER IF NOT EXISTS update_modification_time
62 AFTER UPDATE ON storage
63 FOR EACH ROW
64 BEGIN
65 UPDATE storage
66 SET modification_time = CURRENT_TIMESTAMP
67 WHERE id = OLD.id;
68 END;
69 """)
70 mydb.commit()
71 logger.info("Database setup complete and ready for service")

145 """
