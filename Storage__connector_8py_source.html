86def search(search_term: str) -> List[StorageItem]:
87 """
88 Searches the storage database for entries that match the given search term.
89 Args:
90 search_term: The term to search for in the database. The term will be split
91 into individual words,
92 and each word will be used to search the 'type', 'name', and
93 'info' columns.
94 Returns:
95 A list of tuples containing the rows from the database that match the search criteria.
96 Returns None if there is an SQLite programming error.
97 Raises:
98 sqlite3.ProgrammingError: If there is an error executing the query.
99 """
100 with Session(engine) as session:
101 statement = select(StorageItem)
102 search_words = search_term.split()
103
104 for word in search_words:
105 pattern = f"%{word}%"
106 statement = statement.where(
107 or_(
108 StorageItem.type.like(pattern),
109 StorageItem.name.like(pattern),
110 StorageItem.info.like(pattern)
111 )
112 )
113
114 results = session.exec(statement).all()
115 return list(results)
