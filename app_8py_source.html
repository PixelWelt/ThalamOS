1"""
2StorageManager Application
3
4This module sets up a Flask web application for managing storage items, controlling WLED devices,
5and interacting with a WiFi scale. It provides various routes for rendering templates,
6handling item creation, deletion, and search, as well as retrieving environment configurations
7and scale weight.
8
9Routes:
10 - /: Renders the search page.
11 - /toggleLight: Toggles the power state of the WLED device and renders the search template.
12 - /createItem: Renders the template for creating a new item.
13 - /sendCreation: Handles the creation of a new item by processing the incoming JSON request data.
14 - /item/<item_id>: Handles the request to display an item.
15 - /item/<item>/delete: Deletes an item using the StorageConnector and renders the search.html template.
16 - /search/<term>: Searches for a term in the storage and returns the results in JSON format.
17 - /config/env: Retrieves the environment configuration.
18 - /wifiscale/weight: Retrieves the weight of the scale.
19
20Error Handling:
21 - handle_exception: Handles exceptions by passing through HTTP errors.
22
23Setup:
24 - Initializes the Flask app and sets up CORS.
25 - Loads environment variables from a .env file.

27
28Usage:

89 """
90 Handle the creation of a new item by processing the incoming JSON request data.
91 The function expects a JSON payload with the following structure:
92 {
93 "info": <json>,
94 "type": <str>,
95 "name": <str>,
96 "position": <str>
97 }
98 It extracts the necessary information from the JSON payload and attempts to create a new item
99 using the StorageConnector.CreateItem method.
100 If an exception occurs during the creation process,
101 it prints the exception.
102 Returns:
103 tuple: A dictionary with a status message and an HTTP status code.
104 """
105
106 data = request.get_json()
107 logger.info(f"Received creation request with data: {data}")
108 info = json.dumps(data["info"])
109 typ = data["type"]
110 name = data["name"]
111 pos = data["position"]
112 try:
113 StorageConnector.create_item(pos=pos, typ=typ, name=name, json_data=info)
114 except Exception as e:
115 logger.error(f"Failed to create item with name: {name}, type: {typ}, position: {pos}. Error: {e}")
116 return {"status": "error"}, 500
117 return {"status": "created"}, 201

120@app.route('/item/<item_id>')
121def item(item_id) -> Annotated[str, "item page as a rendered template"]:
122 """
123 Handles the request to display an item.
124 This function performs the following steps:
125 1. Changes the power state of the WLED device to on.
126 2. Fetches the item details from the storage using the provided item identifier.
127 3. Sets the color position on the WLED device based on the fetched item details.
128 4. If the fetched item contains additional information,
129 it parses the data and renders the 'item.jinja2' template
130 with the item details and information.
131 6. If the fetched item does not contain additional information,
132 it renders the 'item.jinja2' template with only the item details.
133 Args:
134 item_id (int): The id of the item to display.
135 Returns:
136 The rendered HTML template for the item.
137 """
138 wledRequests.change_power_state(True)
139 item_sql = StorageConnector.fetch_item(item_id)
140 wledRequests.color_pos(item_sql[1])
141 logger.info(f"Fetched item details for item_id {item_id}: {item_sql}")
142 if item_sql[4]:
143 json_info = json.loads(item_sql[4])
144 return render_template("item.jinja2", item=item_sql, json=json_info, id=item_id)
145
146 return render_template("item.jinja2", item=item_sql, id=item_id)
