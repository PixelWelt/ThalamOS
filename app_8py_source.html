119@app.route('/item/<item_id>')
120def item(item_id) -> str:
121 """
122 Handles the request to display an item.
123 This function performs the following steps:
124 1. Changes the power state of the WLED device to on.
125 2. Fetches the item details from the storage using the provided item identifier.
126 3. Sets the color position on the WLED device based on the fetched item details.
127 4. If the fetched item contains additional information,
128 it parses the data and renders the 'item.jinja2' template
129 with the item details and information.
130 6. If the fetched item does not contain additional information,
131 it renders the 'item.jinja2' template with only the item details.
132 Args:
133 item_id (int): The id of the item to display.
134 Returns:
135 The rendered HTML template for the item.
136 """
137 wledRequests.change_power_state(True)
138 item_sql = StorageConnector.fetch_item(item_id)
139 wledRequests.color_pos(item_sql[1])
140 print(item_sql)
141 if item_sql[4]:
142 json_info = json.loads(item_sql[4])
143 return render_template("item.jinja2", item=item_sql, json=json_info, id=item_id)
144
145 return render_template("item.jinja2", item=item_sql, id=item_id)
